    # Tabs 标签页

    选项卡切换组件。

    <er-tabs :selected.sync="selectedTab">
  <er-tabs-header>
    <er-tabs-item name="tab1">
      tab1
    </er-tabs-item>
    <er-tabs-item name="tab2">
      tab2
    </er-tabs-item>
    <er-tabs-item name="tab3">
      tab3
    </er-tabs-item>
    <er-tabs-item name="tab4">
      tab4
    </er-tabs-item>
  </er-tabs-header>
  <er-tabs-body>
    <er-tabs-panel name="tab1">
      のおれが见栄张って
    </er-tabs-panel>
    <er-tabs-panel name="tab2">
     「世界旅行したい」とつぶやいた
    </er-tabs-panel>
    <er-tabs-panel name="tab3">
      チャーハンを半分ずつしながら
    </er-tabs-panel>
    <er-tabs-panel name="tab4">
      君は小さな声で「连れてって」
    </er-tabs-panel>
  </er-tabs-body>
</er-tabs>

    # 引入方式

    <script>
import { Tabs, TabsHeader, TabsItem, TabsBody, TabsPanel } from 'vue-erui'
export default {
  components: {
    'er-tabs': Tabs,
    'er-tabs-header': TabsHeader,
    'er-tabs-item': TabsItem,
    'er-tabs-body': TabsBody,
    'er-tabs-panel': TabsPanel
  }
}
</script>
