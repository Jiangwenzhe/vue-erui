    # 按钮

    按钮的各式形态： 有基本按钮，图标按钮，加载按钮，禁用按钮 

    <er-button primary>基本按钮</er-button>
<er-button>默认按钮</er-button>
<er-button icon="settings">设置按钮</er-button>
<er-button icon="settings" iconPosition="right">icon右置</er-button>
<er-button :loading="true">正在加载</er-button>
<er-button disabled>禁用按钮</er-button>

    # 引入方式

    <script>
import { Button, ButtonGroup } from 'vue-erui'
export default {
  components: {
    'er-button': Button,  //按钮
    'er-button-group' : ButtonGroup  //按钮组
  }
}
</script>
